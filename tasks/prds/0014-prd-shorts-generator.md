const PB_URL = process.env.POCKETBASE_URL || 'http://localhost:8090';

async function fetchPhotos(options = {}) {
  const { limit = 50, since = null } = options;

  let filter = '';
  if (since) {
    filter = `created >= "${since}"`;
  }

  const response = await fetch(
    `${PB_URL}/api/collections/photos/records?` +
    `sort=-created&perPage=${limit}&filter=${encodeURIComponent(filter)}`
  );

  const data = await response.json();

  return data.items.map(item => ({
    id: item.id,
    title: item.title,
    imageUrl: `${PB_URL}/api/files/photos/${item.id}/${item.image}`,
    thumbnailUrl: item.thumbnail
      ? `${PB_URL}/api/files/photos/${item.id}/${item.thumbnail}`
      : null,
    created: item.created
  }));
}

async function downloadImage(photo, destPath) {
  const response = await fetch(photo.imageUrl);
  const buffer = await response.arrayBuffer();
  await fs.promises.writeFile(destPath, Buffer.from(buffer));
  return destPath;
}
